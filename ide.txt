  Nguyên nhân: trong Apps Script bạn đang gọi:

  ngay đầu doPost(). Vì vậy dù Feedback chỉ gửi text, script vẫn đụng DriveApp, thiếu quyền Drive là lỗi luôn, nên Sheet không ghi gì.

  Sửa Apps Script: chỉ gọi Drive khi có ảnh.

  function doPost(e) {
    try {
      const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME);

      if (!sheet) {
        throw new Error('Không tìm thấy sheet: ' + SHEET_NAME);
      }

      const now = new Date();
      const timeText = Utilities.formatDate(now, 'Asia/Ho_Chi_Minh', 'dd/MM/yyyy HH:mm:ss');

      const content =
        e.parameter.content ||
        e.parameter.feedback ||
        e.parameter.message ||
        '';

      let fileName = e.parameter.fileName || '';
      const fileType = e.parameter.fileType || '';
      const fileSize = e.parameter.fileSize || '';
      let imageBase64 = e.parameter.imageBase64 || '';

      let driveUrl = '';

      if (imageBase64) {
        const folder = DriveApp.getFolderById(FOLDER_ID);

        imageBase64 = imageBase64.replace(/^data:.*;base64,/, '');

        if (!fileName) {
          fileName = 'upload-' + now.getTime() + '.jpg';
        }

        const bytes = Utilities.base64Decode(imageBase64);
        const blob = Utilities.newBlob(bytes, fileType || 'image/jpeg', fileName);

        const file = folder.createFile(blob);
        file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      }

      const nextRow = sheet.getLastRow() + 1;

      sheet.getRange(nextRow, 1, 1, 2).setValues([[content, timeText]]);
      sheet.getRange(nextRow, 6, 1, 4).setValues([[fileName, fileType, fileSize, driveUrl]]);

      return ContentService
        .createTextOutput(JSON.stringify({ ok: true, message: 'Saved successfully', driveUrl }))
        .setMimeType(ContentService.MimeType.JSON);

    } catch (error) {
      return ContentService
        .createTextOutput(JSON.stringify({ ok: false, message: error.toString() }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
